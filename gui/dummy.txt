    def _migrate_schedule_data(self):
        """One-time migration of old schedule_str to new schedules list."""
        migrated = False
        for item in self.valves + self.aux_controls:
            if "schedule_str" in item and item["schedule_str"]:
                migrated = True
                schedule_str = item.pop("schedule_str")
                skip_rainy = item.pop("schedule_skip_rainy", False)
                item.setdefault("schedules", [])
                
                try:
                    new_schedule = None
                    # Try parsing cycle format first
                    if schedule_str.startswith("CYCLE"):
                        match = re.match(r"CYCLE:\s*ON\s*(\d+)\s*m,\s*OFF\s*(\d+)\s*m,\s*x(\d+|∞)\s*at\s*(\d{2}:\d{2})", schedule_str)
                        if match:
                            on_dur, off_dur, count_str, time_part = match.groups()
                            new_schedule = {
                                "id": f"sched_{int(time.time() * 1000)}_{random.randint(100, 999)}",
                                "type": "Cycle", "time": time_part, "on_m": int(on_dur),
                                "off_m": int(off_dur), "count": 0 if count_str == "∞" else int(count_str),
                                "skip_rainy": skip_rainy
                            }
                    else:  # Try parsing fixed time format
                        action, time_part = SchedulerWindow._parse_schedule_string(schedule_str)
                        if action and time_part:
                            new_schedule = {
                                "id": f"sched_{int(time.time() * 1000)}_{random.randint(100, 999)}",
                                "type": "Fixed Time", "action": action, "time": time_part,
                                "skip_rainy": skip_rainy
                            }
                    
                    if new_schedule:
                        item["schedules"].append(new_schedule)
                except Exception as e:
                    self.log(f"Could not migrate schedule '{schedule_str}': {e}")
                    item.setdefault("schedules", [])
        
        if migrated:
            self.log("Migrated old schedule data to new format.")
            self.save_state()
    
    def _start_status_dot_animations(self):
        self._animate_status_dots()
    
    def _animate_status_dots(self):
        # This check ensures the loop stops if the UI is destroyed
        if not hasattr(self, 'root') or not self.root.winfo_exists():
            return
        
        for i, valve_data in enumerate(self.valves):
            # Check if the valve card and its label still exist
            if i < len(self.valve_status_labels) and self.valve_status_labels[i].winfo_exists():
                label = self.valve_status_labels[i]
                if valve_data.get("status"):
                    current_fg = str(label.cget("foreground"))
                    pulse_color_1 = "#10b981"  # Modern green
                    pulse_color_2 = "#059669"  # Darker green
                    
                    try:
                        label.config(foreground=pulse_color_2 if current_fg == pulse_color_1 else pulse_color_1)
                    except tk.TclError:
                        # This can happen on theme change, just reset the color
                        label.config(foreground=pulse_color_1)
        
        self.root.after(750, self._animate_status_dots)

    def update_sensor_readings(self):
        """Fetches and updates sensor readings from the hardware manager."""
        # DHT22
        temp_c, humidity = self.hardware.read_dht22()
        if isinstance(temp_c, (float, int)):
            self.sensor_temp_c.set(f"{temp_c:.1f}°C")
            self.sensor_humidity.set(f"{humidity:.1f}%")
        else:
            self.sensor_temp_c.set(str(temp_c))
            self.sensor_humidity.set(str(humidity))
        
        # DHT11
        temp_c_dht11, humidity_dht11 = self.hardware.read_dht11()
        if isinstance(temp_c_dht11, (float, int)):
            self.sensor_temp_c_dht11.set(f"{temp_c_dht11:.1f}°C")
            self.sensor_humidity_dht11.set(f"{humidity_dht11:.1f}%")
        else:
            self.sensor_temp_c_dht11.set(str(temp_c_dht11))
            self.sensor_humidity_dht11.set(str(humidity_dht11))
        
        # Moisture
        moisture_status = self.hardware.read_moisture()
        if moisture_status == "Wet": 
            self.sensor_moisture.set(f"💧 Wet")
        elif moisture_status == "Dry": 
            self.sensor_moisture.set(f"🔥 Dry")
        else: 
            self.sensor_moisture.set(moisture_status)
        
        if hasattr(self, 'root') and self.root.winfo_exists():
            self.root.after(5000, self.update_sensor_readings)

    def turn_all_systems_off(self):
        if not messagebox.askyesno("Confirm Emergency Stop", 
                                  "Turn OFF all valves and auxiliary controls immediately?\n\nSchedules remain unaffected.", 
                                  icon='warning', parent=self.root): 
            return
        
        off_count = 0
        for items, item_type_name in [(self.valves, "Valve"), (self.aux_controls, "Auxiliary Control")]:
            for item_idx, item in enumerate(items):
                if item.get("status"):
                    item["status"] = False
                    self.hardware.set_pin_state(item['pin'], False)
                    if item_type_name == "Valve": 
                        self._update_valve_on_time_end(item_idx)
                    ts = datetime.datetime.now().isoformat(sep=' ', timespec='seconds')
                    if item_type_name == "Valve": 
                        item.setdefault("history", []).append((ts, "Emergency System Stop OFF"))
                    self.log(f"{item_type_name} '{item['name']}' turned OFF by emergency stop.")
                    off_count += 1
        
        if off_count > 0:
            self.save_state()
            self.filter_valves()
            self.update_aux_controls_ui()
            self.update_dashboard()
            self.notify(f"Emergency Stop: {off_count} system(s) turned OFF.", duration=5000)
        else:
            self.notify("Emergency Stop: No systems were active.", duration=3000)

    def add_valves(self):
        """Add new valves to the system."""
        try:
            count = int(self.valve_count_var.get())
            if count < 1 or count > constants.MAX_VALVES:
                raise ValueError("Invalid count")
        except ValueError:
            messagebox.showerror("Invalid Input", f"Please enter a number between 1 and {constants.MAX_VALVES}.")
            return
        
        if len(self.valves) + count > constants.MAX_VALVES:
            messagebox.showerror("Too Many Valves", f"Cannot add {count} valves. Maximum is {constants.MAX_VALVES}.")
            return
        
        available_pins = [pin for pin in constants.GPIO_PINS if pin not in [v['pin'] for v in self.valves]]
        
        if len(available_pins) < count:
            messagebox.showerror("Not Enough Pins", f"Only {len(available_pins)} GPIO pins available.")
            return
        
        for i in range(count):
            pin = available_pins[i]
            valve_name = f"Valve {len(self.valves) + 1}"
            
            # Try to get a plant emoji based on the name
            for plant_keyword, emoji in constants.PLANT_EMOJIS.items():
                if plant_keyword.lower() in valve_name.lower():
                    icon = emoji
                    break
            else:
                icon = "💧"
            
            new_valve = {
                "name": valve_name,
                "pin": pin,
                "status": False,
                "icon": icon
            }
            
            self._initialize_valve_data(new_valve)
            self.valves.append(new_valve)
        
        self.save_state()
        self.filter_valves()
        self.update_dashboard()
        self.notify(f"Added {count} valve(s).")
        self.log(f"Added {count} new valve(s).")
    
    def reset_valves(self):
        """Reset all valves to OFF state."""
        if not messagebox.askyesno("Confirm Reset", "Turn OFF all valves and clear temporary timers?"):
            return
        
        reset_count = 0
        for i, valve in enumerate(self.valves):
            if valve.get("status"):
                valve["status"] = False
                self.hardware.set_pin_state(valve['pin'], False)
                self._update_valve_on_time_end(i)
                reset_count += 1
        
        if reset_count > 0:
            self.save_state()
            self.filter_valves()
            self.update_dashboard()
            self.notify(f"Reset {reset_count} valve(s).")
            self.log(f"Reset {reset_count} valve(s) to OFF state.")
        else:
            self.notify("All valves were already OFF.")
    
    def toggle_valve(self, idx, duration_min=None):
        """Toggle a valve on or off."""
        valve = self.valves[idx]
        
        if valve.get("locked"):
            self.notify(f"'{valve['name']}' is locked.")
            return
        
        new_status = not valve.get("status", False)
        valve["status"] = new_status
        
        self.hardware.set_pin_state(valve['pin'], new_status)
        
        if new_status:
            self._update_valve_on_time_start(idx)
            if duration_min:
                self._schedule_valve_off(idx, duration_min)
        else:
            self._update_valve_on_time_end(idx)
        
        self.save_state()
        self.filter_valves()
        self.update_dashboard()
        
        status_text = "ON" if new_status else "OFF"
        self.notify(f"'{valve['name']}' turned {status_text}.")
        self.log(f"Valve '{valve['name']}' turned {status_text}.")
    
    def _update_valve_on_time_start(self, idx):
        """Update valve timing when turned on."""
        valve = self.valves[idx]
        valve["current_on_start_time"] = time.time()
        
        ts = datetime.datetime.now().isoformat(sep=' ', timespec='seconds')
        valve.setdefault("history", []).append((ts, "Manual ON"))
    
    def _update_valve_on_time_end(self, idx):
        """Update valve timing when turned off."""
        valve = self.valves[idx]
        
        if valve.get("current_on_start_time"):
            duration = time.time() - valve["current_on_start_time"]
            valve["total_on_time_seconds"] = valve.get("total_on_time_seconds", 0) + duration
            valve["last_on_duration_seconds"] = duration
            valve["last_on_timestamp"] = time.time()
            
            # Calculate water usage
            flow_rate = valve.get("flow_rate_lpm", 0.0)
            if flow_rate > 0:
                water_used = (duration / 60) * flow_rate
                valve["estimated_water_usage_liters"] = valve.get("estimated_water_usage_liters", 0.0) + water_used
            
            valve["current_on_start_time"] = None
            
            ts = datetime.datetime.now().isoformat(sep=' ', timespec='seconds')
            valve.setdefault("history", []).append((ts, f"Manual OFF (ran {utils.format_duration(duration)})"))
    
    def _schedule_valve_off(self, idx, duration_min):
        """Schedule a valve to turn off after a duration."""
        valve = self.valves[idx]
        
        def turn_off():
            if valve.get("status"):
                valve["status"] = False
                self.hardware.set_pin_state(valve['pin'], False)
                self._update_valve_on_time_end(idx)
                
                ts = datetime.datetime.now().isoformat(sep=' ', timespec='seconds')
                valve.setdefault("history", []).append((ts, f"Auto OFF (duration {duration_min}m)"))
                
                self.save_state()
                self.filter_valves()
                self.update_dashboard()
                
                self.notify(f"'{valve['name']}' automatically turned OFF after {duration_min} minutes.")
                self.log(f"Valve '{valve['name']}' automatically turned OFF after {duration_min} minutes.")
        
        # Schedule the turn off
        job_id = self.root.after(duration_min * 60 * 1000, turn_off)
        self.scheduled_jobs[f"valve_{idx}_off"] = job_id

    def undo_remove(self):
        """Undo the last valve removal."""
        if not self.undo_stack:
            self.notify("Nothing to undo.")
            return
        
        if len(self.valves) >= constants.MAX_VALVES:
            self.notify(f"Max valves reached. Cannot restore.")
            return
        
        valve_to_restore = self.undo_stack.pop()
        
        # Check if pin is still available
        current_pins = {v['pin'] for v in self.valves}
        if valve_to_restore['pin'] in current_pins:
            available_pins = [p for p in constants.GPIO_PINS if p not in current_pins]
            if not available_pins:
                self.notify(f"Cannot restore '{valve_to_restore['name']}': Pin taken and no free pins.", duration=4000)
                self.undo_stack.append(valve_to_restore)  # Put it back
                return
            valve_to_restore['pin'] = available_pins[0]
        
        self.valves.append(valve_to_restore)
        self._activate_all_schedules()
        self.save_state()
        self.filter_valves()
        self.update_dashboard()
        
        self.notify(f"Valve '{valve_to_restore['name']}' restored.")
        self.log(f"Valve '{valve_to_restore['name']}' restored.")

    def filter_valves(self, _=None):
        """Filter valves based on search criteria."""
        query = self.search_var.get().strip().lower()
        
        if not query:
            self.filtered_valves = self.valves[:]
        else:
            self.filtered_valves = [v for v in self.valves if 
                                   query in v["name"].lower() or
                                   (query == "on" and v.get("status")) or
                                   (query == "off" and not v.get("status")) or
                                   query in v.get("note", "").lower() or
                                   (f"pin:{v['pin']}" == query)]
        
        self.render_valves_grid()
        self.update_dashboard()
    
    def update_dashboard(self):
        """Update the dashboard metrics."""
        if not hasattr(self, 'dash_valves') or not self.dash_valves.winfo_exists():
            return
        
        v_on = sum(1 for v in self.valves if v.get("status"))
        aux_on = sum(1 for a in self.aux_controls if a.get("status"))
        
        self.dash_valves.config(text=f"Valves: {len(self.valves)}")
        self.dash_on.config(text=f"Total ON: {v_on + aux_on} (Valves: {v_on}, Aux: {aux_on})")
        self.dash_logs.config(text=f"Log Entries: {len(self.logs)}")
        
        # Update timer displays
        for v in self.valves:
            if v.get("status") and v.get("current_on_start_time"):
                elapsed = time.time() - v["current_on_start_time"]
                v["timer_var"].set(f"ON for: {utils.format_duration(elapsed)}")
            else:
                v["timer_var"].set("")

        if filepath:
            try:

                    config_data = json.load(f)
                
                # Validate and import data
                if "valves" in config_data:
                    self.valves = config_data["valves"]
                    for v in self.valves:
                        self._initialize_valve_data(v)
                
                if "aux_controls" in config_data:
                    self.aux_controls = config_data["aux_controls"]
                
                if "automation_rules" in config_data:
                    self.automation_rules = config_data["automation_rules"]
                
                if "theme" in config_data:
                    self.theme = config_data["theme"]
                
                if "location" in config_data:
                    self.location = config_data["location"]
                
                self.save_state()
                self.filter_valves()
                self.update_dashboard()
                self.update_aux_controls_ui()
                
                self.notify(f"Configuration imported from {os.path.basename(filepath)}")
                self.log(f"Configuration imported from {filepath}")

    def on_close(self):
        """Handle application close."""
        # Turn off all systems
        for valve in self.valves:
            if valve.get("status"):
                self.hardware.set_pin_state(valve['pin'], False)
        
        for aux in self.aux_controls:
            if aux.get("status"):
                self.hardware.set_pin_state(aux['pin'], False)
        
        # Cancel all pending jobs
        for job_id in self.root.tk.call('after', 'info'):
            self.root.after_cancel(job_id)
        
        # Clean up hardware
        self.hardware.cleanup()
        
        # Close the application
        self.root.destroy()
